# 樹狀圖

```dataviewjs
// 1. 取得庫中所有不重複的標籤並排序
let tags = Object.keys(app.metadataCache.getTags()).sort();

// 2. 建立樹狀結構
let tagTree = {};
tags.forEach(tag => {
    let cleanTag = tag.replace('#', '');
    let parts = cleanTag.split('/');
    let current = tagTree;
    parts.forEach((part, index) => {
        if (!current[part]) {
            current[part] = { 
                _fullTag: '#' + parts.slice(0, index + 1).join('/'),
                _children: {} 
            };
        }
        current = current[part]._children;
    });
});

// 3. 遞迴渲染函數（改用安全節點容器，防止 HTML 原始碼外露）
function renderTree(node, prefix = "", depth = 0) {
    let keys = Object.keys(node);
    
    keys.forEach((key, index) => {
        let fullTag = node[key]._fullTag;
        let children = node[key]._children;
        let hasChildren = Object.keys(children).length > 0;
        let isLast = index === keys.length - 1;
        
        // A. 決定當前標籤的圖示
        let icon = hasChildren ? "📁 " : (depth === 0 ? "📄 " : "🔹 ");
        
        // B. 根據是不是最後一個子節點，決定當前的分支符號
        let branch = depth > 0 ? (isLast ? "└─ " : "├─ ") : "";
        
        // C. 頂層加粗，子層正常
        let textStyle = depth === 0 ? "font-weight: bold; margin-top: 10px;" : "";
        
        // D. 先建立一個這一行的獨立外層容器
        let rowEl = dv.el("div", "", {
            style: `${textStyle} line-height: 1.6; white-space: pre; display: block;`
        });
        
        // E. 依序把文字前綴與標籤節點塞進去（最安全的作法，徹底絕緣 HTML 錯誤）
        rowEl.appendText(`${prefix}${branch}${icon}`);
        
        let linkEl = document.createElement("a");
        linkEl.className = "tag";
        linkEl.href = fullTag;
        linkEl.textContent = key;
        
        rowEl.appendChild(linkEl);
        
        // F. 如果有子節點，計算下一層的前綴，並繼續往下渲染
        if (hasChildren) {
            let nextPrefix = prefix + (depth > 0 ? (isLast ? "    " : "│   ") : "");
            renderTree(children, nextPrefix, depth + 1);
        }
    });
}

// 4. 開始執行
renderTree(tagTree, "");
```

# 未分類標籤
```dataviewjs
// 1. 自動偵測「目前這個檔案」裡面所有的標籤，當作預設備用清單 
const currentFile = dv.current(); 
const myPredefinedTags = currentFile.file.tags || [];
// 2. 透過系統自動撈出目前所有儲存庫中存在的標籤 
const allExistingTags = Object.keys(app.metadataCache.getTags()); 
// 3. 過濾出那些「你臨時創建、但沒有寫在預設清單裡」的新標籤 
const temporaryTags = allExistingTags.filter(tag => !myPredefinedTags.includes(tag)); 
// 4. 渲染到畫面上 
if (temporaryTags.length > 0) { dv.header(3, "⚠️ 臨時創建的標籤（未加入模版）"); dv.list(temporaryTags); } else { dv.paragraph("✅ 目前所有使用的標籤都已妥善記錄在模版中！"); }
```
